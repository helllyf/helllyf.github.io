雷达坐标转换，从RAE转大地BLH坐标系

标签（空格分隔）： 无趣

RAE转xyz，直角坐标系与极坐标系之间的转换

/*雷达坐标系的直角坐标系xyz 转极坐标RAE(m,du,du) */
void xyz2RAE(Real x, Real y, Real z, Real& R, Real& A, Real& E)
{
    R = sqrt(x*x + y*y + z*z);
    A = atan2(y, x)*(180.0/_PI);
    E = acos(sqrt(x*x + y*y)/R)*(180.0/_PI);
}
/*雷达坐标系的极坐标RAE(m,du,du)转直角坐标系xyz */
void RAE2xyz(Real R, Real A, Real E, Real& x, Real& y, Real& z)
{
    //度转弧度
    Real A1  = A*_PI/180.0; 
    Real E1  = E*_PI/180.0; 
    x = R*cos(E1)*cos(A1);
    y = R*cos(E1)*sin(A1);
    z = R*sin(E1);
}

xyz转XYZ，目标相对雷达的直角坐标系转换成相对于大地的直角坐标系

/*矩阵计算辅助函数*/
void MetricMulti3x3( const Real *x, const Real *y, Real *z )
{
    z[0] = x[0]*y[0]+x[1]*y[3]+x[2]*y[6];
    z[1] = x[0]*y[1]+x[1]*y[4]+x[2]*y[7];
    z[2] = x[0]*y[2]+x[1]*y[5]+x[2]*y[8];

    z[3] = x[3]*y[0]+x[4]*y[3]+x[5]*y[6];
    z[4] = x[3]*y[1]+x[4]*y[4]+x[5]*y[7];
    z[5] = x[3]*y[2]+x[4]*y[5]+x[5]*y[8];

    z[6] = x[6]*y[0]+x[7]*y[3]+x[8]*y[6];
    z[7] = x[6]*y[1]+x[7]*y[4]+x[8]*y[7];
    z[8] = x[6]*y[2]+x[7]*y[5]+x[8]*y[8];
}
void MetricMulti33x31( const Real *x, const Real *y, Real *z )
{
    z[0] = x[0]*y[0]+x[1]*y[1]+x[2]*y[2];
    z[1] = x[3]*y[0]+x[4]*y[1]+x[5]*y[2];
    z[2] = x[6]*y[0]+x[7]*y[1]+x[8]*y[2];
}

/*将某点在大地直角坐标系下的XYZ，转化为l雷达坐标系下的xyz*/
/*输入：lon lat hgt 为雷达经纬高， XYZ为目标在大地直角坐标系中的坐标*/
/*输出：xyz为目标在雷达坐标系下的坐标*/
void XYZ2xyz(Real lon, Real lat, Real hgt, Real X, Real Y, Real Z, Real& x, Real& y, Real& z)
{
    //将雷达的GPS转化为大地直角坐标
    Real Rx,Ry,Rz;
    GPS2XYZ(lon, lat, hgt, Rx, Ry, Rz);
    //计算pos-pos0
    Real Tmp[3];
    Tmp[0] = X - Rx;
    Tmp[1] = Y - Ry;
    Tmp[2] = Z - Rz;
    //旋转矩阵
    //Real XX[9] = {1, 0, 0, 0, cos(90-lat) ,sin(90-lat), 0, -sin(90-lat), cos(90-lat)};
    //Real ZZ[9] = {cos(90+lon), sin(90+lon), 0, -sin(90+lon), cos(90+lon), 0, 0, 0, 1};
    Real XX[9] = {1, 0, 0, 0, cos((90-lat)*_PI/180.0) ,-sin((90-lat)*_PI/180.0), 0, sin((90-lat)*_PI/180.0), cos((90-lat))*_PI/180.0};
    Real ZZ[9] = {cos((90+lon)*_PI/180.0), -sin((90+lon)*_PI/180.0), 0, sin((90+lon)*_PI/180.0), cos((90+lon)*_PI/180.0), 0, 0, 0, 1};

    Real Rotate[9];
    MetricMulti3x3(XX, ZZ, Rotate);

    //计算雷达坐标系下的位置
    Real Pos[3];
    MetricMulti33x31(Rotate,Tmp,Pos);

    x = Pos[0];
    y = Pos[1];
    z = Pos[2];
}

/*将某点在雷达坐标系下的xyz转化为大地直角坐标系下的XYZ*/
/*输入：lon lat hgt 为雷达经纬高， xyz为目标在雷达坐标系下的坐标*/
/*输出：*XYZ为目标在大地直角坐标系中的坐标*/
void xyz2XYZ(Real lon, Real lat, Real hgt, Real x, Real y, Real z, Real& X, Real& Y, Real& Z)
{
    //将雷达的GPS转化为大地直角坐标
    Real Rx,Ry,Rz;
    GPS2XYZ(lon, lat, hgt, Rx, Ry, Rz);

    //旋转矩阵
    //Real XX[9] = {1, 0, 0, 0, cos(90-lat) ,-sin(90-lat), 0, sin(90-lat), cos(90-lat)};
    //Real ZZ[9] = {cos(90+lon), -sin(90+lon), 0, sin(90+lon), cos(90+lon), 0, 0, 0, 1};
    Real XX[9] = {1, 0, 0, 0, cos((90-lat)*_PI/180.0) ,-sin((90-lat)*_PI/180.0), 0, sin((90-lat)*_PI/180.0), cos((90-lat))*_PI/180.0};
    Real ZZ[9] = {cos((90+lon)*_PI/180.0), -sin((90+lon)*_PI/180.0), 0, sin((90+lon)*_PI/180.0), cos((90+lon)*_PI/180.0), 0, 0, 0, 1};

    Real Rotate[9];
    MetricMulti3x3(ZZ, XX, Rotate);

    //计算雷达坐标系下的位置
    Real tmp[3];
    Real xyz[3];
    xyz[0] = x;
    xyz[1] = y;
    xyz[2] = z;
    MetricMulti33x31(Rotate,xyz,tmp);

    X = tmp[0] + Rx;
    Y = tmp[1] + Ry;
    Z = tmp[2] + Rz;
}

XYZ转BLH，大地直角坐标系转大地经纬高坐标系

/*将某点在大地坐标系下的GPS，转化为大地直角坐标系下的XYZ*/
/*输入：lon lat hgt 为经纬高，单位为度                    */

void GPS2XYZ(Real lon, Real lat, Real hgt, Real& X, Real& Y, Real& Z)
{
    Real B  = lat*_PI/180.0; // 纬度
    Real L  = lon*_PI/180.0; // 经度
    Real H = hgt; 

#ifdef WGS_84
    Real ac = 6378137.0;
    Real f = 1.0/298.257223563;    
#else
    Real ac = 6378137.0;
    Real f =1.0/298.257222101;
#endif

    Real e_2 = 2*f-f*f;
    Real N =  ac/sqrt(1-e_2*(sin(B)*sin(B)));
    /*地心直角坐标*/
    X = (N+H) * cos(B) * cos(L);
    Y = (N+H) * cos(B) * sin(L);
    Z = (N*(1-e_2)+H)*sin(B);
}
/*将某点在大地直角坐标系下的XYZ，转化为大地坐标系下的GPS*/
/*迭代算法*/
void XYZ2GPS(Real X, Real Y, Real Z, Real& lon, Real& lat, Real& hgt)
{
#ifdef WGS_84
    Real ac = 6378137.0;
    Real f = 1.0/298.257223563;    
#else
    Real ac = 6378137.0;
    Real f =1.0/298.257222101;
#endif

    //Real e_2 = 2*f-f*f;
    Real bc = (1 -f)*ac;
    Real e_2 = (ac*ac-bc*bc)/(ac*ac);
    //初值
    Real Rm0 = ac;
    Real Hm0 = sqrt(X*X + Y*Y + Z*Z) - sqrt(ac*bc);
    Real Bm0 = atan2( Z*(Rm0 + Hm0), (Rm0*(1-e_2)+Hm0)*sqrt(X*X + Y*Y));
    //迭代算法
    Real N = 0;
    Real Hm = 0;
    Real Bm = 0;    
    while(fabs(Bm-Bm0)>1e-5)
    {
        N = ac/sqrt(1-e_2*pow(sin(Bm0) ,2));
        Hm = sqrt(X*X+Y*Y)/cos(Bm0) - N;
        Bm = atan2(Z*(N+Hm), sqrt(X*X + Y*Y)*(N*(1-e_2)+Hm));
        Bm0 = Bm;
    }
    lat = Bm/(_PI/180.0);
    lon = atan2(Y, X)/(_PI/180.0);
    hgt = sqrt(X*X+Y*Y)/cos(Bm) - ac/sqrt(1-e_2*pow(sin(Bm) ,2));//重新计算
}
